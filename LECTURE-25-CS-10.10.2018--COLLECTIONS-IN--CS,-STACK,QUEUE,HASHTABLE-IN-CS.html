using System;
using System.Collections;
class Program
{
    public static void Main()
    {

        Hashtable ht = new Hashtable();

        ht.Add(1, 'A'); // key , value pair , keys can be anything ...int, char, string etc.
        ht.Add(7, 'B'); // key , value pair , keys can any combination of int, char etc
        ht.Add(15, 'C'); // key , value pair, keys can be in any sequennce
        ht.Add(22, 'D'); // key , value pair, keys are not index

        Console.WriteLine(ht[1]);
        Console.WriteLine(ht[7]);
        Console.WriteLine(ht[91]);
        Console.WriteLine(ht[22]);
        Console.WriteLine(ht[15]);
    }
}
