Program-7: Manager and subordinate program - object oriente approach

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter First Number :");
        int n1 = int.Parse(Console.ReadLine());

        Console.Write("Enter Second Number :");
        int n2= int.Parse(Console.ReadLine());

        A a = new A();
        a.add(n1,n2);
    }    
}
class A
{
    public void add(int a,int b)
    {
        int result = a + b;
        Console.WriteLine(result);
    }
}

Program-8: Manager and subordinate program - object oriente approach - right approach

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter First Number :");
        int n1 = int.Parse(Console.ReadLine());

        Console.Write("Enter Second Number :");
        int n2= int.Parse(Console.ReadLine());

        A a = new A();
        int result= a.add(n1,n2);
        Console.WriteLine(result);

    }    
}
class A
{
    public int add(int a,int b)
    {
        int result = a + b;
        return result;       
    }
