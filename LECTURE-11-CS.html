Program-4 : parameter passing ( one argument supply)

using System;

class Program
{
    static void Main()
    {

        Console.Write("Please enter you name: ");
        string name1= Console.ReadLine();
        SayHello(name1);       
    }

    static void  SayHello(string name2)
    {
        Console.WriteLine("Hello "+name2);
    }
}

Program - 5: actual and formal name of parameter can be same

using System;

class Program
{
    static void Main()
    {

        Console.Write("Please enter you name: ");
        string name= Console.ReadLine();
        SayHello(name);       // actual parameter
    }

    static void  SayHello(string name)// formal
    {
        Console.WriteLine("Hello "+name);
    }
}


Program-6: supply two arguments to a function and return the processed result.

using System;

class Program
{
    static void Main()
    {

        Console.Write("Please first number : ");
        int a =int.Parse( Console.ReadLine());
        Console.Write("Please second number : ");
        int b = int.Parse(Console.ReadLine());

        int result = Addition(a,b);

        Console.WriteLine("Addition =  " + result);

    }

    static int  Addition(int a, int b)
    {
        return a + b;
      
    }
}

Program-7: scope of local variables - local to block only bydefault

using System;

class Program
{
    static void Main()
    {
        int a = 10;
        Console.WriteLine("value of a before funciton calling ="+a);
        display(a);
        Console.WriteLine("value of a after funciton calling =" + a);

    }
    static void  display(int a)
    {
        a = a + 10;
        Console.WriteLine("value of a within function body =" + a);
    }
}
