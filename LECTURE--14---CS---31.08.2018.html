Program-1: method overloading - method signature

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter First Number :");
        int n1 = int.Parse(Console.ReadLine());

        Console.Write("Enter Second Number :");
        int n2 = int.Parse(Console.ReadLine());

        Console.Write("Enter Second Number :");
        int n3 = int.Parse(Console.ReadLine());

        A a = new A();

        int result1 = a.add(n1, n2);
        Console.WriteLine("Addition of first two numbers = " + result1);

        int result2 = a.add(n1, n2, n3);
        Console.WriteLine("Addition of three numbers = " + result2);
    }
}
class A
{
    public int add(int a, int b)
    {
        int result = a + b;
        return result;
    }
    public int add(int a, int b, int c)
    {
        int result = a + b + c;
        return result;
    }
}
