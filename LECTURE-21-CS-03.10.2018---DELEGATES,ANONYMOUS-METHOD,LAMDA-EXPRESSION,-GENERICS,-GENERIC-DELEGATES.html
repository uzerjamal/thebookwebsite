Program-4: GENERIC METHODS

// advantage -2 : no boxing is happening

}

Program-1: Generic delegate
// FUNC  generic delegate is used for parameterized methods
// Func is for int, double, etc
using System;
class Program
{
    static void Main()
    {
        Func<int,int,int> mydel = A.sum; // first two are int parameter and last int is retrun type
        int result=mydel(2, 3);
        Console.WriteLine("addition ="+result);
        Console.ReadKey();
    }
}
class A
{
    public static int sum(int a, int b)
    {
        return (a + b);
    }
}
