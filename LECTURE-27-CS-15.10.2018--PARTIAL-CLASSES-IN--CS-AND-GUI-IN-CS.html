Program-4: partial keyword is used to have two concrete classes in the same same namespace with same name

using System;

namespace ns
{
    class Program
    {
        static void Main()
        {
            A x = new A();
            x.Display1();
            x.Display2();
            Console.ReadKey();
        }
    }
    partial class A
    {
        public void Display1()
        {
            Console.WriteLine(" Display 1");
        }
    }

    partial class A
    {
        public void Display2()
        {
            Console.WriteLine(" Display 2");
        }
    }
}
