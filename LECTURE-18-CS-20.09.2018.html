Program-17: Indexer - 1 : using setter and getter for array

using System;

public class A
{
    private int[] a = { 101, 102, 103, 104, 105 };
   
    public void  SetValue(int index, int val)// setter method
    {
            a[index ]= val;
    }
    public int  GetValue(int index)// getter method
    {
         return a[index];
    }
}

class Program
{
    static void Main()
    {

        A x = new A();
        x.SetValue(2,202);
        Console.WriteLine("value - after " +x.GetValue(2));
        Console.ReadKey();
    }
}

Program-19: Indexer - 3: using indexer for 2D array

using System;

public class A
{
    private int[,] matrix ={
                     {101,102,103,104,105},
                     {201,202,203,204,205}};

    public int this[int id1,int id2]
    {
        set
        {
            matrix[id1,id2] = value;
        }
        get
        {
            return matrix[id1,id2];
        }
   
    }
 
}

class Program
{
    static void Main()
    {

        A a = new A();             
        Console.WriteLine("value - before "+a[1,2]);
        a[1,2] = 222222;
        Console.WriteLine("value - after " +a[1,2]);
        Console.ReadKey();
    }
}
