Program-1:ArrayList-1: we can add different types of objects in collection ( array list)

using System;
using System.Collections;
class Program
{
    public static void Main()
    {
        ArrayList arraylist = new ArrayList();// non - generic ArrayList

        arraylist.Add(101);// we can add integer values , auto boxing
        arraylist.Add(10.2);// we can double type values,auto boxing
        arraylist.Add('A'); // we can add char type values,auto boxing
        arraylist.Add("abc"); // we can add string type,auto boxing
        arraylist.Add(new object()); // we can add object type      

    }
}
