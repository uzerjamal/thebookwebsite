Program-13:

1. how to pass variable number of arugments to a function in c#
2. what is params

using System;
class Program
{
    static void Main()
    {
        A a = new A();
        int result = a.add(1,2,3,4,5);
        Console.WriteLine("Addition of three numbers = " + result);
        Console.ReadKey();
    }
}
class A
{
    public int add(params int[] s)
    {
        int result = 0;

        for (int i = 0; i < s.Length; i++)
        {
            result = result +s[i];
        }
        return result;
    }
}
