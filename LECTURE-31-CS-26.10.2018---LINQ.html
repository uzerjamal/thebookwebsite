Program-10: order by LINQ for list of students
// Collection: List type
using System;
using System.Linq;
using System.Collections;
using System.Collections.Generic;
class Program
{
    public static void Main()
    {
        List<Student> list= new List<Student> ();

        Student s1 = new Student() { id=  103, name="ramesh"};
        Student s2 = new Student() { id = 101, name = "suresh" };
        Student s3 = new Student() { id = 102, name = "raju" };

        list.Add(s1);
        list.Add(s2);
        list.Add(s3);
       
        IEnumerable<Student> result = from x in list
                                      orderby x.id
                                      select x;

        foreach (Student y in result)
        {
            Console.WriteLine("ID = "+y.id + "Name= " +y.name);
        }

    }
}

class Student
{
    public int id { get; set; }
    public string name { get; set; }

Program-16:sum of salary of those employee whose name endswith "sh" using  Where and Sum Extended function

using System;
using System.Linq;
using System.Collections;
using System.Collections.Generic;
class Program
{
    public static void Main()
    {
        List<Employee> list = new List<Employee>();

        Employee e1 = new Employee() { id = 103, name = "ramesh",salary=85000 };
        Employee e2 = new Employee() { id = 101, name = "suresh", salary = 90000 };
        Employee e3 = new Employee() { id = 102, name = "raju", salary = 95000 };

        list.Add(e1);
        list.Add(e2);
        list.Add(e3);

        int result = list.Where(x=>x.name.EndsWith("sh")).Sum(x => x.salary);
                    // chaining of extension method
        Console.WriteLine("sum of salary of all employee ="+result);
       

    }
}

class Employee
{
    public int id { get; set; }
    public string name { get; set; }
    public int salary { get; set; }
}
