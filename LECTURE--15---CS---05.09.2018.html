Program-1: use of this keyword

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter First Number :");
        int n1 = int.Parse(Console.ReadLine());

        Console.Write("Enter Second Number :");
        int n2 = int.Parse(Console.ReadLine());

        A a = new A(n1, n2);

        int result1 = a.add();
        Console.WriteLine("Addition  = " + result1);

        int result2 = a.sub();
        Console.WriteLine("Subtraction  = " + result2);

        int result3 = a.mul();
        Console.WriteLine("Multiplication  = " + result3);

        int result4 = a.div();
        Console.WriteLine("Division  = " + result4);


    }
}
class A
{
    int a, b;

    public A(int a, int b)
    {
        this.a = a;
        this.b = b;
    }
    public int add()
    {
        int result = a + b;
        return result;
    }
    public int sub()
    {
        int result = a - b;
        return result;
    }
    public int mul()
    {
        int result = a * b;
        return result;
    }
    public int div()
    {
        int result = a / b;
        return result;
    }
}

using System;

class Program
{

   {

   }
}

{

Program-4: Data Encapsulation and logic encapsulation and abstraction

using System;

class Program
{
   static void Main(string[] args)
   {

       Console.WriteLine("enter a numbr : ");

       int n = int.Parse(Console.ReadLine());

       Square obj = new Square(n);

       int sq = obj.getSquare();

       Console.WriteLine("Square of given number : " + sq);

       Console.ReadKey();

   }
}

class Square
{
   int x, sq;  // data encapsulation

   public Square(int x)
   {
       this.x = x;
   }
   public int getSquare() // interface or abstract form of logic or abstraction
   {
       sq = calculateSquare();
       return sq;
   }
   private int calculateSquare() // logic encapsulation
   {
       return (x * x);
   }
}
