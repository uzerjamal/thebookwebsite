Program-1: more than one variables can be initialized out side the function body

using System;

class Program
{
    static void Main()
    {
        int a;
        int b;
        Display(out a,out b);
        Console.WriteLine("value of a after funciton calling display =" + a);
        Console.WriteLine("value of b after funciton calling display =" + b);
    }
    static void Display(out int a,out int b)
    {
        a = 10;
        b = 20;
        Console.WriteLine("value of a within function body =" + a);
        Console.WriteLine("value of b within function body =" + b);
    }   
}
